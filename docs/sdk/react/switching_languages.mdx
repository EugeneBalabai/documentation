For language management, use `useTolgee` hook. With passed [`language`](../api/core_package/events) event, so the component will get re-rendered when this event occurs.

```jsx
import { useTolgee } from '@tolgee/react';

export const LangSelector = () => {
  const tolgee = useTolgee(['language']);

  return (
    <select
      onChange={(e) => tolgee.changeLanguage(e.target.value)}
```
